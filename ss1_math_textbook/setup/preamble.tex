% --- Beautiful Boxes (tcolorbox) ---
\usepackage[many,breakable]{tcolorbox}

% Example Box
\newtcolorbox[auto counter, number within=chapter]{example}[1][]{
    enhanced,
    breakable,
    colback=examplebg,
    colframe=primary,
    fonttitle=\bfseries,
    title=Example \thetcbcounter \ifstrempty{#1}{}{ --- #1},
    attach boxed title to top left={yshift=-2mm, xshift=5mm},
    boxed title style={colback=primary}
}

% Definition Box
\newtcolorbox{definition}[1][]{
    enhanced,
    breakable,
    colback=secondary!20,
    colframe=secondary!80!black,
    fonttitle=\bfseries,
    coltitle=black,
    title=Definition\ifstrempty{#1}{}{: #1},
    arc=0mm,
    leftrule=4mm
}

% Exercise Box
\newtcolorbox{exercisebox}[1][]{
    enhanced,
    breakable,
    colback=white,
    colframe=accent,
    fonttitle=\bfseries,
    coltitle=black,
    title=Exercise\ifstrempty{#1}{}{~#1},
    attach boxed title to top center={yshift=-2mm},
    boxed title style={colback=accent}
}

% Note Box
\newtcolorbox{notebox}[1][Note]{
    enhanced,
    colback=notebg,
    colframe=accent!80!black,
    fonttitle=\bfseries\small,
    coltitle=black,
    title=#1,
    arc=3mm,
    leftrule=3mm,
    top=4pt, bottom=4pt
}
